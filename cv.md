---
function mergeArrays(arr1, arr2) {

  if (arr1 === undefined && arr2 === undefined){
    return arr1
  }
  
  const arr3 = [...arr1, ...arr2];
  const result = Array.from(new Set(arr3));
  return result.sort((a,b) => a-b)
  
}
